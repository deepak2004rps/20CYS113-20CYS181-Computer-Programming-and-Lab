#include<stdio.h>
int isprime(int n)
{
    int i,flag=0;
    for(i=2;i<n;i++)
    {
        if(n%i==0)
        {
            flag+=1;
        }
    }
        if(flag==0)
        return 1;
        else
        return 0;
}
generateprime(int s,int e)
{
    int flag;
    for(int i=s;i<=e;i++)
    {
        flag=isprime(i);
        if(flag==1)
        {
            printf("%d ",i);
        }
    }
}
int main()
{   int s,e;
    scanf("%d %d",&s,&e);
    printf("Prime numbers between %d and %d are: ",s,e);
    generateprime(s,e);
    return 0;
}
