#include<stdio.h>
int main()
{
    int n;
    scanf("%d",&n);
    int v[n];
    v[n-1]=1;
    for(int i=0;i<n-1;i++)
    {
        v[i]=0;
    }
    for(int i=0;i<n;i++)
    {
        for(int k=0; k<n; k++)
{        if(v[k]==0)
        {
            printf(" ");
        }
        else
        {
            printf("%d ",v[k]);
        }
}    for(int j=0;j<n-1;j++)
    {
        v[j] = v[j] + v[j+1];
    }
    printf("\n");
    }
return 0;
}
